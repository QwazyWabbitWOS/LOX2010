w140   
alias zoomcontrol "alias upkey zoomup;alias downkey zoomdown;echo zoom control;alias controlswitch wacontrol" 
alias wacontrol "alias upkey waup;alias downkey wadown;echo wide angle control;alias controlswitch zoomcontrol" 
zoomcontrol 
bind z +zoom     // binds zoom feature to "Z" 
bind q +wa  // binds wide angle to "q" 
bind < upkey     // binds decrease zoom / wide angle to "<" 
bind > downkey   // binds increase zoom / wide angle to ">" 
bind / controlswitch  // toggles "<" and ">" keys between zoom and wide angle control
